904 LBox boundingBox() const noexcept
905 {
906 const LPoint &p { pos() };
907 const LPoint &s { size() };
908
909 LBox box
910 {
911 p.x(),
912 p.y(),
913 p.x() + s.w(),
914 p.y() + s.h(),
915 };
916
917 LBox childBox;
918
919 for (LView *child : children())
920 {
921 if (!child->mapped())
922 continue;
923
924 childBox = child->boundingBox();
925
926 if (childBox.x1 < box.x1)
927 box.x1 = childBox.x1;
928
929 if (childBox.y1 < box.y1)
930 box.y1 = childBox.y1;
931
932 if (childBox.x2 > box.x2)
933 box.x2 = childBox.x2;
934
935 if (childBox.y2 > box.y2)
936 box.y2 = childBox.y2;
937 }
938
939 return box;
